\bbox
\begin{thm}{\textbf{Transformation Theorem}}
\bi
\ii Let $Y=G(X)$ be a random variable that is a function of the random variable $X$.
\ii If we assume $G(X)$ is a monotonic, then it has an inverse $X = G^{-1}(Y)=H(Y)$.
\ii We can find the pdf and cdf of $Y$ as follows:
\[ F_Y(y) = F_X(H(y)) \ \ , \mbox{and by the chain rule} \]
\[ f_Y(y) = f_X(H(y)) \cdot H'(y).\]
\ei
\end{thm}
